```python
#!/usr/bin/env python
# -*- coding: utf-8 -*-
# @Author: _Tao
# @Time: 19-5-22 22:33


class Node(object):

	def __init__(self, val):
		self.val = val
		self.next = None

	def __str__(self):
		return str(self.val)

	__repr__ = __str__


class HashTable(object):
	def __init__(self, init_length=15):
		# 长度
		self.__table = [None] * init_length
		# 当前表中数据个数
		self.__count = 0

	def insert(self, val):
		# 插入

		# 索引
		index = self.hash_index(val)

		if not self.__table[index]:
			self.__table[index] = Node(val)
		else:
			head = self.__table[index]
			while head.next:
				head = head.next

			# 添加到链表尾部
			head.next = Node(val)

		# 长度+1
		self.count += 1

		# 计算是否达到限制容量
		if self.need_rehash():
			self.rehash()

	def delete(self, val):
		# 删除值

		index = self.hash_index(val)
		# 得到头结点
		node = self.__table[index]
		# 保存前一个节点
		pre = None
		while node:
			if node.val == val:
				# 如果pre为空,即val是头结点的值
				if not pre:
					# 要删除的是头结点,那么将头结点的下一个节点放到表中
					self.__table[index] = node.next
				else:
					pre.next = node.next
				self.count -= 1
				# 删除
				del node
				break
			pre = node
			node = node.next

	def search(self, val):
		# 搜索
		# :param val: 需要搜索的值
		# :return: 如果没有找到,返回None,找到了就返回节点

		index = self.hash_index(val)

		# 得到头结点
		node = self.__table[index]
		while node:
			if node.val == val:
				break
			node = node.next

		return node

	def traverse(self):
		# 遍历数据

		for node in self.__table:
			if node:
				# 遍历一条链表的数据
				while node:
					print(node.val, end=' ')
					node = node.next
				print()

	@property
	def count(self):
		# 当前数据个数

		return self.__count

	@count.setter
	def count(self, count):
		self.__count = count

	def table_length(self):
		# 表总长度

		return len(self.__table)

	def __used_index(self):
		# 使用了下标的个数,即有多少个索引下标使用了

		length = 0
		for node in self.__table:
			if node:
				length += 1

		return length

	def rehash(self):
		# 当达到限制容量时,扩容后重新赋值

		# 临时表,扩容使用,扩容倍数*2
		tmp = [None] * (self.table_length() * 2)
		# 读取头节点
		for node in self.__table:
			if node:
				# 头节点中的值
				val = node.val
				# 计算索引
				index = self.hash_index(val, tmp)
				# 将头节点放到临时表中新的索引处
				tmp[index] = node

		# 删除表后,将临时表重新赋值
		del self.__table
		self.__table = tmp

	def need_rehash(self):
		# 是否需要扩容,如果已经使用超过70%容量,就需要扩容

		return self.__used_index() / self.table_length() >= 0.7

	def hash_index(self, val, table=None):
		# 计算索引
		# :param val: 要计算的值
		# :param table: 表,有可能是临时表(扩容中用上)
		
		# 直接使用hash算法,就不用自己写了
		# 先转化成str类型,避免某些list之类的不可hash导致报错
		# 除以总长度得到索引位置
		return int(hash(str(val))) % len(table if table else self.__table)


if __name__ == '__main__':
	table = HashTable(10)
	import random

	vals = [random.randint(1, 100) for _ in range(100)]
	for val in vals:
		table.insert(val)

	print('--->遍历')
	table.traverse()

	print('--->长度')
	print(table.count)
	print(table.table_length())

	print('--->搜索')
	print(table.search(vals[0]))

	table.delete(vals[0])
	print('--->删除后的长度')
	print(table.count)
	print(table.table_length())
```

```python
class Node(object):

		return str(self.value)

	def __init__(self, init_length=15):
		self.__table = [None] * init_length
		# 列表中总共右多少数据
		self.__size = 0

	def __setitem__(self, key, value):
		self.__insert(key, value)

	def __getitem__(self, item):
		return self.__search(item)

	def __delitem__(self, key):
		self.__delete(key)

	def __iter__(self):
		return self

	def __next__(self):
		# 迭代器
		if not hasattr(self, 'index'):
			self.index = -1
		self.index += 1

		if self.index < len(self.__table):
			node = self.__table[self.index]
			values = []
			if node:
				while node:
					values.append(node.value)
					node = node.next

			return values
		else:
			raise StopIteration

	def __insert(self, key, value):
		# 插入

		# 获取索引
		index = self.__hash_index(key)

		if not self.__table[index]:
			self.__table[index] = Node(key, value)
		else:
			node = self.__table[index]
			while node.next:
				# 如果key值相同了，就说明值需要覆盖
				if node.key == key:
					node.value = value
					break
				node = node.next
			# 如果是正常退出，就追加到链表结尾
			else:
				# 添加到链表尾部
				node.next = Node(key, value)

		# 长度+1
		self.__size += 1

		# self.__rehash()

		index = self.__hash_index(key)

		# 得到头结点
		node = self.__table[index]
		while node:
			if node.key == key:
				break
			node = node.next
		# 如果正常退出，就说明没有找到

		while node:
			if node.key == key:
				# 如果pre为空,表示要删除的点时头结点

					self.__table[index] = node.next
```
