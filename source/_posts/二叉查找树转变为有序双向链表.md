一道微软的面试题: 输入一棵二元查找树，将该二元查找树转换成一个排序的双向链表。

```python
#!/usr/bin/env python
# -*- coding: utf-8 -*-
# @Author: _Tao
# @Time: 19-5-24 22:40


import random


class Node(object):

	def __init__(self, value, left=None, right=None):
		self.value = value
		self.left = left
		self.right = right


class Tree2DoubleLink(object):

	def __init__(self):
		self.root = None

	def insert(self, value):
		if not self.root:
			self.root = Node(value)
		else:
			self.__insert(value, self.root)

	def __insert(self, value, node):
		if not node:
			node = Node(value)
		elif value <= node.value:
			node.left = self.__insert(value, node.left)
		elif value > node.value:
			node.right = self.__insert(value, node.right)

		return node

	def mid_order(self, node):
		# 中序遍历
		if node:
			self.mid_order(node.left)
			print(node.value, end=' ')
			self.mid_order(node.right)

	def tree_to_link(self, node):
		# 将二叉排序树转成双向链表
		# 添加
		if not hasattr(self, 'head'):
			# 头结点
			self.head = None
			# 移动的节点
			self.cur = None

		# 中序遍历
		if node:
			self.tree_to_link(node.left)
			# 创建头结点
			if not self.head:
				self.head = node
			# 移动的节点
			elif not self.cur:
				self.cur = node
				# 头结点指向第二个节点
				self.head.right = self.cur
				# 第二个节点的左指针指向头节点
				self.cur.left = self.head
			else:
				# 前一个节点的右指针指向当前节点
				self.cur.right = node
				# 当前节点的左指针指向前一个节点
				node.left = self.cur
				# 替换
				self.cur = node
			self.tree_to_link(node.right)

		return self.head


if __name__ == '__main__':
	tree = Tree2DoubleLink()
	# 随机输入
	for i in [random.randint(0, 100) for _ in range(10)]:
		tree.insert(i)

	# 中序遍历
	tree.mid_order(tree.root)
	print('\n')

	# 转成双向链表
	head = tree.tree_to_link(tree.root)
	# 从头节点开始输出
	while head and head.right:
		print(head.value, end=' ')
		head = head.right
	print(head.value)

	# 从尾节点开始输出
	tail = head
	while tail:
		print(tail.value, end=' ')
		tail = tail.left
```
