思路:
如果两个单链表有交叉节点,那么他们的尾节点一定是相同的,所以可以计算出链表的长度,然后判断长短,将长度较长的链表的头节点移动(长-短)个节点,然后遍历之后的节点.

```python
class Node(object):

    def __init__(self, val):
        self.val = val
        self.next = None

    def __str__(self):
        return self.val


def get_communal_node(head1, head2):
    # 获取交叉节点
    length1= get_length(head1)
    lenght2 = get_length(head2)

    # 如果链表1的长度大于链表2
    # 那么就将链表1的头节点移动length1-length2个位置
    if length1 > lenght2:
        while lenght2 < length1:
            head1 = head1.next
            lenght2 += 1
    # 同上
    elif lenght2 > length1:
        while length1 < lenght2:
            head2 = head2.next
            length1 += 1

    # 从同一起点位置开始移动
    while (head1 and head2):
        # 如果值相等(实际上应该判断是否是同一个节点),就说明这个是交叉的节点
        if head1.val == head2.val:
            return head1

        head1 = head1.next
        head2 = head2.next
    
    return None


def get_length(head):
    # 获取链表长度
    length = 0
    # head = copy.deepcopy(_head)
    while head:
        length += 1
        head = head.next

    return length


def get_head(links):
    # 构建链表,返回头节点
    head = None
    cur = None
    for val in links:
        node = Node(val)
        if not head:
            head = node
            cur = head
        else:
            cur.next = node
            cur = cur.next

    return head


if __name__ == '__main__':
    head1 = get_head(['java', 'world', 'node', 'B+', 'x', 'y', 'z'])
    head2 = get_head(['ab', 'bc', 'ef', 'hello', 'python', 'x', 'y', 'z'])

    node = get_communal_node(head1, head2)
    print(node)
```
